# Mission to set up trackable values used in elections.
# Popularity of each bloc from 1 to 100 can be altered by events or missions. Is used at election time to determine who wins. If it drops too low can trigger an event to change leadership.
mission "Republic Election Set Up"
	landing
	invisible
	on offer
		"terms" = 1
		"incumbent popularity" = 50
		"opposition popularity" = 50
		fail
# For testing only, to be removed.
		dialog "Republic Election initialised"

# Election missions which trigger on 1st September every fifth year (or whenever the player lands after that date).
# Measures relative popularity of each bloc, with a small advantage for the incumbent.
# The longer a bloc is in power the lower their popularity will go.
# If a new bloc gains power they get a bump in popularity and the outgoing bloc ?
mission "Republic Election Incumbent Wins"
	invisible
	repeat
	to offer
		"day" == 1
		"month" == 9
		"year" %5 == 0
		or
			"incumbent popularity" >= "opposition popularity"
	on offer
		"terms" ++
		"incumbent popularity" - (10 * "terms")
		"opposition popularity" + 10
		# Dialog for testing purposes, to be replaced with something more suitable.
		dialog
			"The $<incumbent> coalition has won another term in office."
		fail

mission "Republic Election Challenger Wins"
	invisible
	repeat
	to offer
		"day" == 1
		"month" == 9
		"year" %5 == 0
		and
			"opposition popularity" > "incumbent popularity"
	on offer
		"republic bloc" 3 = "republic bloc" 1
		"republic bloc" 1 = "republic bloc" 2
		"republic bloc" 2 = "republic bloc" 3
		"republic bloc" 3 = "Null"
		"political leader" 3 = "political leader" 1
		"political leader" 1 = "political leader" 2
		"political leader" 2 = "political leader" 3
		"political leader" 3 = "Null"
		"terms" = 1
		"transition popularity" = "incumbent popularity"
		"incumbent popularity" = "opposition popularity" - 10
		"opposition popularity" = "transition popularity" + 10
		clear "transition popularity"
		# Dialog for testing purposes, to be replaced with something more suitable.
		dialog
			"The $<incumbent> coalition has won, replacing $<opposition> to govern the Republic."
		fail

# Checks the level of popularity of the Opposition every six months. If it is too low a new leader is announced and the bloc gains a boost in popularity.
mission "Republic Opposition New Leader"
	invisible
	repeat
	to offer
		"day" == 22
		or
			"month" == 3
			"month" == 9
		and
			"opposition popularity" <= 30
	on offer
		"political leader" 2 = "politician"
		"opposition popularity" + 10
		# Dialog for testing purposes, to be replaced with something more suitable.
		dialog
			"After a recent bad run, the $<opposition> group have decided that a change is in order and have elected a new leader from amongst their ranks; $<opposition leader>."


# Checks the level of popularity of the Incumbents every six months. If it is too low a vote of no confidence is announced.
mission "Republic Opposition New Leader"
	invisible
	repeat
	to offer
		"day" == 22
		or
			"month" == 6
			"month" == 12
		and
			"incumbent popularity" <= 20
	on offer
		# Dialog for testing purposes, to be replaced with something more suitable.
		dialog
			"The $<incumbent> coalition, worried about a recent bad showing in the polls and protests on several key worlds, have announced that they will be enacting a vote of no confidence in current Prime Minister, $<prime minister>. The Prime Minister was unavailable for comment."
