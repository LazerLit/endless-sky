# Mission to set up trackable values used in elections.
# Popularity of each bloc from 1 to 100 can be altered by events or missions. Is used at election time to determine who wins. If it drops too low can trigger an event to change leadership.
mission "Republic Election Set Up"
	landing
	invisible
	on offer
		set string "bloc 1" "Moderates"
		set string "bloc 2" "Traditionalists"
		set string "leader 1" "Arthur Baker"
		set string "leader 2" "Mustafa Kikelomo"
		set "terms"
		"bloc in power" = 1
		"incumbent popularity" = 50
		"opposition popularity" = 50
		fail
# For testing only, to be removed.
		dialog "Republic Election initialised"

# Election missions which trigger on 1st September every fifth year (or whenever the player lands after that date).
# Measures relative popularity of each bloc, with a small advantage for the incumbent.
# The longer a bloc is in power the lower their popularity will go.
# If a new bloc gains power they get a slight dip in popularity and the outgoing bloc gets a bump.
mission "Republic Election Bloc 1 Elected"
	invisible
	repeat
	to offer
		"day" == 1
		"month" == 9
		"year" %5 == 0
		"bloc in power" == 2
		"incumbent popularity" < "opposition popularity"
	on offer
		set "terms"
		"bloc in power" = 1
		"transition popularity" = "incumbent popularity"
		"incumbent popularity" = "opposition popularity"
		"opposition popularity" = "transition popularity"
		clear "transition popularity"
		# Dialog for testing purposes, to be replaced with something more suitable.
		dialog
			"The <incumbent> coalition has won the election."
		fail

mission "Republic Election Bloc 1 Re-elected"
	invisible
	repeat
	to offer
		"day" == 1
		"month" == 9
		"year" %5 == 0
		"bloc in power" == 1
		"incumbent popularity" >= "opposition popularity"
	on offer
		"terms" ++
		"incumbent popularity" -= (10 * "terms")
		"opposition popularity" += 10
		# Dialog for testing purposes, to be replaced with something more suitable.
		dialog
			"The <incumbent> coalition has won another term."
		fail

# Checks the level of popularity of the Opposition every six months. If it is too low a new leader is announced and the bloc gains a boost in popularity.
mission "Republic Opposition New Leader"
	invisible
	repeat
	to offer
		"day" == 22
		or
			"month" == 3
			"month" == 9
		"opposition popularity" <= 30
	on offer
		set string "leader 2" "politician"
		"opposition popularity" + 10
		# Dialog for testing purposes, to be replaced with something more suitable.
		dialog
			"After a recent bad run, the <opposition> group have decided that a change is in order and have elected a new leader from amongst their ranks; <opposition leader>."
